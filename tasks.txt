fid = fopen(filename, 'r');

if fid == -1, error('Dosya açılamadı'); end

fseek(fid, 0, 'eof'); file_size = ftell(fid); frewind(fid);

Ns_total = floor(file_size / (4 * 2));

s_dat_total = fread(fid, [2, Ns_total], 'float32');

fclose(fid);

    s_dat_total = s_dat_total.';
